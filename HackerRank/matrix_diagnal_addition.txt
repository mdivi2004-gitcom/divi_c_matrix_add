#include<stdio.h>
#include<stdlib.h>
int main()
{
    int i,j,i1,j1,i2,r=0,l=0,n,d;
    scanf("%d",&n);
    int a[n][n];
    for(i1=0;i1<i;i1++)
    {
        for(j1=0;j1<i1;j1++)
        {
         scanf("%d",&a[i1][j1])   ;
        }
    
    }    
    for(i2=n;i2>0;i2--)
    {
        r=r+a[i2][i2];
        l=l+a[i2][n-1-i2]        ;
    }
    
    d=abs(r-l);
    printf("%d",d);
    return 0;
}
